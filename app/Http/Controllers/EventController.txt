<?php
use App\Models\PostKajianAcara;
use Illuminate\Http\Request;

class EventController extends Controller
{
    public function fetch(Request $request)
    {
        $start = $request->query('start');
        $end   = $request->query('end');

        // Ambil data dari database sesuai rentang tanggal
        $events = PostKajianAcara::whereBetween('event_date', [$start, $end])->get();

        // Format data agar cocok dengan script JS
        $formatted = $events->map(function ($event) {
            return [
                'id'        => $event->id,
                'title'     => $event->event_name,
                'start'     => $event->event_date . 'T' . ($event->event_time ?? '00:00:00'),
                'speaker'   => $event->speaker,
                'location'  => $event->location,
                'is_routine'=> $event->is_routine ?? false,
            ];
        });

        return response()->json($formatted);
    }
}
